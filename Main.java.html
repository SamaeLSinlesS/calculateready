14       public static String parse(String expression) throws Exception {
15           int num1;
16           int num2;
17           String oper;
18           String result;
19           boolean isRoman;
20           String[] operands = expression.split("[+\\-*/]");
21           if (operands.length != 2) throw new Exception("Должно быть два операнда");
22           oper = detectOperation(expression);
23           if (oper == null) throw new Exception("Неподдерживаемая математическая операция");
24           //если оба числа римские
25           if (Roman.isRoman(operands[0]) && Roman.isRoman(operands[1])) {
26               num1 = Roman.convertToArabian(operands[0]);
27               num2 = Roman.convertToArabian(operands[1]);
28               isRoman = true;
29           }
30           //если оба числа арабские
31           else if (!Roman.isRoman(operands[0]) && !Roman,isRoman(operands[1])) {
32               num1 = Integer.parseInt(operands[0]);
33               num2 = Integer.parseInt(operands[1]);
34               isRoman = false;
35           }
36           //если одно число римское, а другое - арабское
37           else {
38               throw new Exception("Числа должны быть в одном формате")
39           }
40           if (num1 > 10 || num2 > 10) {
41               throw new Exception("Числа должны быть от 1 до 10");
42           }
43           int arabian = calc(num1, num2, oper);
44           if (isRoman) {
45               //если римское число получилось меньше, либо равно нулю - генерируем ошибку
46               if (arabian <= 0) {
47                   throw new Exception("Римское число должно быть больше нуля");
48               }
49               //Конвертируем результаты операции из арабского в римское
50               result = Roman.convertToRoman(arabian);
51           } else {
52               //Конвертируем арабское число в тип String
53               result = String.valueOf(arabian);
54           }
55           //возвращаем результат
56           return result;
57   }

59       Static String detectOperation(String expression) {
60           if (expression.contains("+")) return "+";
61           else if (expression.contains("-")) return "-";
62           else if (expression.contains("*")) return "*";
63           else if (expression.contains("/")) return "/";
64           else return null;
65   
66       }

68       static int calc(int a, int b, String oper) {
69   
70           if (oper.equals("+")) return a + b;
71           else if (oper.equals("-")) return a - b;
72           else if (oper.equals("*")) return a * b;
73           else return a / b;
74       }

78   class Roman {
79       static String[] romanArray = new String[]{"0", "I", "II", "III", "IV", "V", "VI", "VII", "VIII", "IX", "X", "XI",
80               "XII", "XIII", "XIV", "XV", "XVI", "XVII", "XVIII", "XIX", "XX"};

82       public static boolean isRoman(String val) {
83           for (int i = 0; i < romanArray.length; i++) {
84               if (val.equals(romanArray[i])) {
85                   return true;
86               }
87           }
88           return false;
89       }
90       public static int convertToArabian(String roman) {
91           for (int i = 0; i < romanArray.length; i++) {
92               if (roman.equals(romanArray[i])) {
93                   return i;
94               }
95           }
96           return -1;
97       }
98   
99       public static String convertToRoman(int arabian) { return romanArray[arabian]; }
